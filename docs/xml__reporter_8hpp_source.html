75  void report_testcase(intern::testcase const& tc_) override {
76  *this << intern::fmt::XSPACE << "<testcase name=\"" << tc_.name() << "\" classname=\""
77  << tc_.context() << "\" time=\"" << tc_.duration() << "\"";
78  switch (tc_.state()) {
79  case intern::testcase::result::ERROR:
80  *this << ">" << intern::fmt::LF << intern::fmt::XSPACE << intern::fmt::SPACE
81  << "<error message=\"" << tc_.reason() << "\"></error>" << intern::fmt::LF;
82  if (m_capture) {
83  print_system_out(tc_);
84  }
85  *this << intern::fmt::XSPACE << "</testcase>";
86  break;
87  case intern::testcase::result::FAILED:
88  *this << ">" << intern::fmt::LF << intern::fmt::XSPACE << intern::fmt::SPACE
89  << "<failure message=\"" << tc_.reason() << "\"></failure>"
90  << intern::fmt::LF;
91  if (m_capture) {
92  print_system_out(tc_);
93  }
94  *this << intern::fmt::XSPACE << "</testcase>";
95  break;
96  default:
97  if (m_capture) {
98  *this << ">" << intern::fmt::LF;
99  print_system_out(tc_);
100  *this << intern::fmt::XSPACE << "</testcase>";
101  } else {
102  *this << "/>";
103  }
104  break;
105  }
106  *this << intern::fmt::LF;
107  }
