102  void
103  report_testcase(testcase const& tc_) override {
104  *this << fmt::XSPACE << "<testcase name=\"" << tc_.name() << "\" classname=\"" << tc_.suite_name()
105  << "\" time=\"" << tc_.duration() << "\"";
106  switch (tc_.state()) {
107  case testcase::result::ERROR:
108  *this << ">" << fmt::LF << fmt::XSPACE << fmt::SPACE << "<error message=\"" << tc_.reason()
109  << "\"></error>" << fmt::LF;
110  if (m_capture) {
111  print_system_out(tc_);
112  }
113  *this << fmt::XSPACE << "</testcase>";
114  break;
115  case testcase::result::FAILED:
116  *this << ">" << fmt::LF << fmt::XSPACE << fmt::SPACE << "<failure message=\"" << tc_.reason()
117  << "\"></failure>" << fmt::LF;
118  if (m_capture) {
119  print_system_out(tc_);
120  }
121  *this << fmt::XSPACE << "</testcase>";
122  break;
123  default:
124  if (m_capture) {
125  *this << ">" << fmt::LF;
126  print_system_out(tc_);
127  *this << fmt::XSPACE << "</testcase>";
128  } else {
129  *this << "/>";
130  }
131  break;
132  }
133  *this << fmt::LF;
134  }
