102  void
103  report_testcase(testcase const& tc_) override {
104  *this << fmt::SPACE << tc_.name() << " (" << tc_.duration() << "ms)" << fmt::LF << fmt::XSPACE;
105  if (m_capture) {
106  *this << "stdout = \"" << tc_.cout() << "\"" << fmt::LF << fmt::XSPACE;
107  *this << "stderr = \"" << tc_.cerr() << "\"" << fmt::LF << fmt::XSPACE;
108  }
109  switch (tc_.state()) {
110  case testcase::result::ERROR: *this << colored(RED) << "ERROR! " << tc_.reason(); break;
111  case testcase::result::FAILED: *this << colored(BLUE) << "FAILED! " << tc_.reason(); break;
112  case testcase::result::PASSED: *this << colored(GREEN) << "PASSED!"; break;
113  default: break;
114  }
115  *this << colored() << fmt::LF;
116  }
117 
118  void
119  end_report() override {
120  if (m_abs_errs > 0) {
121  *this << colored(YELLOW);
122  } else if (m_abs_fails > 0) {
123  *this << colored(BLUE);
124  } else {
125  *this << colored(CYAN);
126  }
127  *this << "=== Result ===" << fmt::LF << "passes: " << m_abs_tests - m_abs_fails - m_abs_errs << "/"
128  << m_abs_tests << " failures: " << m_abs_fails << "/" << m_abs_tests << " errors: " << m_abs_errs << "/"
129  << m_abs_tests << " (" << m_abs_time << "ms)" << colored() << fmt::LF;
130  }
