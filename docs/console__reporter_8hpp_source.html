114  void report_testcase(intern::testcase const& tc_) override {
115  *this << intern::fmt::SPACE << "Run Testcase [" << tc_.name() << "](" << tc_.suite_name()
116  << "); time = " << tc_.duration() << "ms" << intern::fmt::LF << intern::fmt::XSPACE;
117  if (m_capture) {
118  *this << "stdout = \"" << tc_.cout() << "\"" << intern::fmt::LF << intern::fmt::XSPACE;
119  *this << "stderr = \"" << tc_.cerr() << "\"" << intern::fmt::LF << intern::fmt::XSPACE;
120  }
121  switch (tc_.state()) {
122  case intern::testcase::result::ERROR:
123  *this << (m_color ? intern::fmt::ANSI_MAGENTA : "") << "ERROR! " << tc_.reason();
124  break;
125  case intern::testcase::result::FAILED:
126  *this << (m_color ? intern::fmt::ANSI_RED : "") << "FAILED! " << tc_.reason();
127  break;
128  case intern::testcase::result::PASSED:
129  *this << (m_color ? intern::fmt::ANSI_GREEN : "") << "PASSED!";
130  break;
131  default: break;
132  }
133  *this << (m_color ? intern::fmt::ANSI_RESET : "") << intern::fmt::LF;
134  }

146  << "/" << m_abs_tests << " ; time = " << m_abs_time << "ms"
147  << (m_color ? intern::fmt::ANSI_RESET : "") << intern::fmt::LF;
