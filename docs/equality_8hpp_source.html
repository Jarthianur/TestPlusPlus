73  template<typename V, typename E = V>
74  auto
75  operator()(V const& actual_value, E const& expected_value) const -> comparison {
76  static_assert(SCTF_INTERN_IS_FLOAT(V) && SCTF_INTERN_IS_FLOAT(E),
77  "The floating point comparator must not be used with other types than float, or double!");
78 
79  typename std::decay<V>::type epsilon_ = static_cast<V>(m_eps);
80  return (std::abs(actual_value - expected_value) <=
81  std::max(std::abs(actual_value), std::abs(expected_value)) * epsilon_) != m_neg ?
82  comparison() :
83  comparison(m_neg ? m_neg_cmp_str : m_cmp_str,
84  std::forward_as_tuple(to_string(actual_value), to_string(expected_value)));
85  }
