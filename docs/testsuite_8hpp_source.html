80  virtual void
81  run() {
82  if (m_state != execution_state::DONE) {
83  m_stats.m_num_tests = m_testcases.size();
84  streambuf_proxy buf_cout(std::cout);
85  streambuf_proxy buf_cerr(std::cerr);
86  m_setup_fn();
87  std::for_each(m_testcases.begin(), m_testcases.end(), [this, &buf_cerr, &buf_cout](testcase& tc_) {
88  if (tc_.state() == testcase::result::NONE) {
89  m_pretest_fn();
90  tc_();
91  switch (tc_.state()) {
92  case testcase::result::FAILED: ++m_stats.m_num_fails; break;
93  case testcase::result::ERROR: ++m_stats.m_num_errs; break;
94  default: break;
95  }
96  m_exec_dur += tc_.duration();
97  m_posttest_fn();
98  tc_.cout(buf_cout.str());
99  tc_.cerr(buf_cerr.str());
100  buf_cout.clear();
101  buf_cerr.clear();
102  }
103  });
104  m_teardown_fn();
105  m_state = execution_state::DONE;
106  }
107  }
