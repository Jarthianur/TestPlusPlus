78  virtual void run() {
79  if (m_state != execution_state::DONE) {
80  m_setup_fn();
81  m_stats.m_num_tests = m_testcases.size();
82  streambuf_proxy buf_cout(std::cout);
83  streambuf_proxy buf_cerr(std::cerr);
84  std::for_each(m_testcases.begin(), m_testcases.end(),
85  [this, &buf_cerr, &buf_cout](testcase& tc_) {
86  if (tc_.state() == testcase::result::NONE) {
87  m_pretest_fn();
88  tc_();
89  switch (tc_.state()) {
90  case testcase::result::FAILED: ++m_stats.m_num_fails; break;
91  case testcase::result::ERROR: ++m_stats.m_num_errs; break;
92  default: break;
93  }
94  m_exec_dur += tc_.duration();
95  m_posttest_fn();
96  tc_.cout(buf_cout.str());
97  tc_.cerr(buf_cerr.str());
98  buf_cout.clear();
99  buf_cerr.clear();
100  }
101  });
102  m_teardown_fn();
103  m_state = execution_state::DONE;
104  }
105  }
