46 template<typename T>
47 static std::string const& name_for_type() {
48  static thread_local std::string name;
49  if (name.length() > 0) {
50  return name;
51  }
52 #ifdef SCTF_SYS_UNIX
53  std::string const sig(__PRETTY_FUNCTION__);
54  auto const b = sig.rfind("T = ") + 4;
55  name = sig.substr(b, sig.find_first_of(";]", b) - b);
56  name.erase(std::remove(name.begin(), name.end(), ' '), name.end());
57 #else
58  std::string const sig(typeid(T).name());
59  auto b = sig.find("struct ");
60  if (b != std::string::npos) {
61  name = sig.substr(b + 7);
62  return name;
63  }
64  b = sig.find("class ");
65  if (b != std::string::npos) {
66  name = sig.substr(b + 6);
67  } else {
68  name = std::move(sig);
69  }
70 #endif
71  return name;
72 }

77 inline std::string escaped_char(char c_) {
78  char const* ret;
79  switch (c_) {
80  case '\r': ret = "\\r"; break;
81  case '\n': ret = "\\n"; break;
82  case '\t': ret = "\\t"; break;
83  case '\f': ret = "\\f"; break;
84  case '\v': ret = "\\v"; break;
85  case '\"': ret = "\\\""; break;
86  default: return std::string(1, c_);
87  }
88  return ret;
89 }

94 static std::string escaped_string(std::string const& str_) {
95  std::string s = str_;
96  std::size_t p = 0;
97  while ((p = s.find_first_of("\r\n\t\f\v\"", p)) != std::string::npos) {
98  s.replace(p, 1, escaped_char(s[p]));
99  p += 2;
100  }
101  return s;
102 }
