49 template<typename T>
50 static std::string const& name_for_type() {
51  static thread_local std::string name;
52  if (name.length() > 0) {
53  return name;
54  }
55 #ifdef SCTF_SYS_UNIX
56  std::string const sig(__PRETTY_FUNCTION__);
57  auto const b = sig.rfind("T = ") + 4;
58  name = sig.substr(b, sig.find_first_of(";]", b) - b);
59  name.erase(std::remove(name.begin(), name.end(), ' '), name.end());
60 #else
61  std::string const sig(typeid(T).name());
62  auto b = sig.find("struct ");
63  if (b != std::string::npos) {
64  name = sig.substr(b + 7);
65  return name;
66  }
67  b = sig.find("class ");
68  if (b != std::string::npos) {
69  name = sig.substr(b + 6);
70  } else {
71  name = std::move(sig);
72  }
73 #endif
74  return name;
75 }

82 inline std::string escaped_char(char c_) {
83  char const* ret;
84  switch (c_) {
85  case '\r': ret = "\\r"; break;
86  case '\n': ret = "\\n"; break;
87  case '\t': ret = "\\t"; break;
88  case '\f': ret = "\\f"; break;
89  case '\v': ret = "\\v"; break;
90  case '\"': ret = "\\\""; break;
91  default: return std::string(1, c_);
92  }
93  return ret;
94 }

101 static std::string escaped_string(std::string const& str_) {
102  std::string s = str_;
103  std::size_t p = 0;
104  while ((p = s.find_first_of("\r\n\t\f\v\"", p)) != std::string::npos) {
105  s.replace(p, 1, escaped_char(s[p]));
106  p += 2;
107  }
108  return s;
109 }
