52  std::time_t stamp = std::chrono::system_clock::to_time_t(ts_->timestamp());
53  char buff[128];
54  std::strftime(buff, 127, "%FT%T", std::localtime(&stamp));
55  *this << SCTF_SPACE << "<testsuite id=\"" << m_id++ << "\" name=\"" << ts_->name()
56  << "\" errors=\"" << ts_->statistics().errors() << "\" tests=\""
57  << ts_->statistics().tests() << "\" failures=\"" << ts_->statistics().failures()
58  << "\" skipped=\"0\" time=\"" << ts_->execution_time() << "\" timestamp=\"" << buff
59  << "\">" << SCTF_LF;

64  void report_testcase(private_::testcase const& tc_) override {
65  *this << SCTF_XSPACE << "<testcase name=\"" << tc_.name() << "\" classname=\""
66  << tc_.context() << "\" time=\"" << tc_.duration() << "\"";
67  switch (tc_.state()) {
68  case private_::testcase::result::ERROR:
69  *this << ">" << SCTF_LF << SCTF_XSPACE << SCTF_SPACE << "<error message=\""
70  << tc_.err_msg() << "\"></error>" << SCTF_LF;
71  if (m_capture) {
72  print_system_out(tc_);
73  }
74  *this << SCTF_XSPACE << "</testcase>";
75  break;
76  case private_::testcase::result::FAILED:
77  *this << ">" << SCTF_LF << SCTF_XSPACE << SCTF_SPACE << "<failure message=\""
78  << tc_.err_msg() << "\"></failure>" << SCTF_LF;
79  if (m_capture) {
80  print_system_out(tc_);
81  }
82  *this << SCTF_XSPACE << "</testcase>";
83  break;
84  default:
85  if (m_capture) {
86  *this << ">" << SCTF_LF;
87  print_system_out(tc_);
88  *this << SCTF_XSPACE << "</testcase>";
89  } else {
90  *this << "/>";
91  }
92  break;
93  }
94  *this << SCTF_LF;
95  }
