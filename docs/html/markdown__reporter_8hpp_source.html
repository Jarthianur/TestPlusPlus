65  void report_testcase(private_::testcase const& tc_) override {
66  char const* status = "";
67  switch (tc_.state()) {
68  case private_::testcase::result::ERROR: status = "ERROR"; break;
69  case private_::testcase::result::FAILED: status = "FAILED"; break;
70  case private_::testcase::result::PASSED: status = "PASSED"; break;
71  default: break;
72  }
73  *this << "|" << tc_.name() << "|" << tc_.context() << "|" << tc_.duration() << "ms|"
74  << status << "|";
75  if (m_capture) {
76  print_system_out(tc_.cout());
77  print_system_out(tc_.cerr());
78  }
79  *this << SCTF_LF;
80  }

93  void print_system_out(std::string const& out_) {
94  std::string line;
