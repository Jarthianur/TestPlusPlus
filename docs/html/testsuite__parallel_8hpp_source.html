60  void run() override {
61  if (m_state != execution_state::DONE) {
62  if (m_testcases.size() > static_cast<std::size_t>(std::numeric_limits<long>::max())) {
63  throw std::overflow_error("Too many testcases! Size would overflow loop variant.");
64  }
65  SCTF_PRIVATE_EXEC_SILENT(m_setup_fn)
66  const long tc_size = static_cast<long>(m_testcases.size());
67  m_stats.m_num_of_tests = m_testcases.size();
68  streambuf_proxy_omp mt_buf_cout(std::cout);
69  streambuf_proxy_omp mt_buf_cerr(std::cerr);
70 
71 #pragma omp parallel
72  { // BEGIN parallel section
73  double tmp = 0.0;
74  std::size_t fails = 0;
75  std::size_t errs = 0;
76  // OpenMP 2 compatible - MSVC not supporting higher version
77 #pragma omp for schedule(dynamic)
78  for (long i = 0; i < tc_size; ++i) {
79  auto& tc = m_testcases[static_cast<std::size_t>(i)];
80  if (tc.state() == testcase::result::NONE) {
81  SCTF_PRIVATE_EXEC_SILENT(m_pretest_fn)
82  tc();
83  switch (tc.state()) {
84  case testcase::result::FAILED: ++fails; break;
85  case testcase::result::ERROR: ++errs; break;
86  default: break;
87  }
88  tmp += tc.duration();
89  SCTF_PRIVATE_EXEC_SILENT(m_posttest_fn)
90  tc.cout(mt_buf_cout.str());
91  tc.cerr(mt_buf_cerr.str());
92  mt_buf_cout.clear();
93  mt_buf_cerr.clear();
94  }
95  }
96 #pragma omp critical
97  { // BEGIN critical section
98  m_stats.m_num_of_fails += fails;
99  m_stats.m_num_of_errs += errs;
100  m_execution_time = std::max(m_execution_time, tmp);
101  } // END critical section
102  } // END parallel section
103  SCTF_PRIVATE_EXEC_SILENT(m_teardown_fn)
104  m_state = execution_state::DONE;
105  }
106  }
