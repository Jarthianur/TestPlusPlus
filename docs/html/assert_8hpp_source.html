224 template<typename T>
225 static void assert_throws(void_function&& fn_, char const* tname_, loc const& loc_) {
226  try {
227  fn_();
228  } catch (T const&) {
229  return;
230  } catch (std::exception const& e) {
231  throw assertion_failure("Wrong exception thrown, caught " + to_string(e), loc_);
232  } catch (...) {
233  throw assertion_failure("Wrong exception thrown", loc_);
234  }
235  throw assertion_failure(std::string("No exception thrown, expected ") + tname_, loc_);
236 }

264 static void assert_runtime(void_function&& fn_, double max_ms_, loc const& loc_) {
265  try {
266  duration dur;
267  fn_();
268  double dur_ms = dur.get();
269  if (dur_ms > max_ms_) {
270  throw assertion_failure("runtime > " + to_string(max_ms_) + "ms", loc_);
271  }
272  } catch (std::exception const& e) {
273  throw assertion_failure(e.what(), loc_);
274  } catch (...) {
275  throw assertion_failure("Unknown exception thrown", loc_);
