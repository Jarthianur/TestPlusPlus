131 static void assert_statement(V const& val_, E const& expect_, C&& cmp_, code_location const& loc_) {
132  comparison res = cmp_(val_, expect_);
133  if (!res) {

138 template<typename T>
139 static void assert_throws(void_function&& fn_, code_location const& loc_) {
140  try {
141  fn_();
142  } catch (T const&) {
143  return;
144  } catch (std::exception const& e) {
145  throw assertion_failure("Wrong exception thrown, caught " + to_string(e), loc_);
146  } catch (...) {
147  throw assertion_failure("Wrong exception thrown", loc_);
148  }
149  throw assertion_failure("No exception thrown, expected " + name_for_type<T>(), loc_);
150 }

152 static void assert_nothrow(void_function&& fn_, code_location const& loc_) {
153  try {
154  fn_();
155  } catch (const std::exception& e) {
156  throw assertion_failure("Expected no exception, caught " + to_string(e), loc_);
157  } catch (...) {
158  throw assertion_failure("Expected no exception", loc_);
159  }
160 }

162 static void assert_runtime(void_function&& fn_, double max_ms_, code_location const& loc_) {
163  try {
164  duration dur;
165  fn_();
166  double dur_ms = dur.get();
167  if (dur_ms > max_ms_) {
168  throw assertion_failure("runtime > " + to_string(max_ms_) + "ms", loc_);
169  }
170  } catch (std::exception const& e) {
171  throw assertion_failure(e.what(), loc_);
172  } catch (...) {
173  throw assertion_failure("Unknown exception thrown", loc_);
174  }
175 }
