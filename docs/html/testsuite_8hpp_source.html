71  virtual void run() {
72  if (m_state != execution_state::DONE) {
73  m_setup_fn();
74  m_stats.m_num_of_tests = m_testcases.size();
75  streambuf_proxy buf_cout(std::cout);
76  streambuf_proxy buf_cerr(std::cerr);
77  std::for_each(m_testcases.begin(), m_testcases.end(),
78  [this, &buf_cerr, &buf_cout](testcase& tc_) {
79  if (tc_.state() == testcase::result::NONE) {
80  m_pretest_fn();
81  tc_();
82  switch (tc_.state()) {
83  case testcase::result::FAILED:
84  ++m_stats.m_num_of_fails;
85  break;
86  case testcase::result::ERROR: ++m_stats.m_num_of_errs; break;
87  default: break;
88  }
89  m_exec_dur += tc_.duration();
90  m_posttest_fn();
91  tc_.cout(buf_cout.str());
92  tc_.cerr(buf_cerr.str());
93  buf_cout.clear();
94  buf_cerr.clear();
95  }
96  });
97  m_teardown_fn();
98  m_state = execution_state::DONE;
99  }
100  }
