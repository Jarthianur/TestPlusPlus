228 template<typename T, typename F>
229 auto
230 assert_throws(F&& fn_, char const* tname_, loc const& loc_) -> T {
231  try {
232  fn_();
233  } catch (T const& e) {
234  return e;
235  } catch (std::exception const& e) {
236  throw assertion_failure(
237  "Wrong exception thrown, caught " + to_string(e) + "(\"" + escaped_string(e.what()) + "\")", loc_);
238  } catch (...) {
239  throw assertion_failure("Wrong exception thrown", loc_);
240  }
241  throw assertion_failure(std::string("No exception thrown, expected ") + tname_, loc_);
242 }
