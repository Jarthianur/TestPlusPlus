222 template<typename T>
223 static void assert_throws(void_function&& fn_, char const* tname_, loc const& loc_) {
224  try {
225  fn_();
226  } catch (T const&) {
227  return;
228  } catch (std::exception const& e) {
229  throw assertion_failure("Wrong exception thrown, caught " + to_string(e), loc_);
230  } catch (...) {
231  throw assertion_failure("Wrong exception thrown", loc_);
232  }
233  throw assertion_failure(std::string("No exception thrown, expected ") + tname_, loc_);
234 }

262 static void assert_runtime(void_function&& fn_, double max_ms_, loc const& loc_) {
263  try {
264  duration dur;
265  fn_();
266  double dur_ms = dur.get();
267  if (dur_ms > max_ms_) {
268  throw assertion_failure("runtime > " + to_string(max_ms_) + "ms", loc_);
269  }
270  } catch (std::exception const& e) {
271  throw assertion_failure(e.what(), loc_);
272  } catch (...) {
273  throw assertion_failure("Unknown exception thrown", loc_);
274  }
