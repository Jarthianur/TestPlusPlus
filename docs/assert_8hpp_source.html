233 template<typename T, typename F>
234 auto assert_throws(F&& fn_, char const* tname_, loc const& loc_) -> T {
235  try {
236  fn_();
237  } catch (T const& e) {
238  return e;
239  } catch (std::exception const& e) {
240  throw assertion_failure("Wrong exception thrown, caught " + to_string(e), loc_);
241  } catch (...) {
242  throw assertion_failure("Wrong exception thrown", loc_);
243  }
244  throw assertion_failure(std::string("No exception thrown, expected ") + tname_, loc_);
245 }
