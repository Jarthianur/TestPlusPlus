90  void report_testcase(intern::testcase const& tc_) override {
91  char const* status = "";
92  switch (tc_.state()) {
93  case intern::testcase::result::ERROR: status = "ERROR"; break;
94  case intern::testcase::result::FAILED: status = "FAILED"; break;
95  case intern::testcase::result::PASSED: status = "PASSED"; break;
96  default: break;
97  }
98  *this << "|" << tc_.name() << "|" << tc_.context() << "|" << tc_.duration() << "ms|"
99  << status << "|";
100  if (m_capture) {
101  print_system_out(tc_.cout());
102  print_system_out(tc_.cerr());
103  }
104  *this << intern::fmt::LF;
105  }
